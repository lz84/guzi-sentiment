## 五、外系统对接设计

### 5.1 API接口设计

#### 核心接口

```yaml
# 舆情数据API
GET /api/v1/sentiment/events
  参数:
    - event_type: election|policy|economic|social
    - time_range: 1h|24h|7d|30d
    - sentiment: positive|negative|neutral
    - min_confidence: 0.0-1.0
  响应:
    {
      "events": [
        {
          "id": "evt_xxx",
          "type": "election",
          "title": "候选人A负面新闻激增",
          "sentiment": {"score": -0.8, "confidence": 0.85},
          "entities": ["候选人A"],
          "impact": {"direction": "negative", "magnitude": "high"},
          "sources": ["twitter", "news"],
          "timestamp": "2026-03-17T06:00:00Z"
        }
      ],
      "summary": {
        "total": 15,
        "by_sentiment": {"positive": 3, "negative": 10, "neutral": 2}
      }
    }

# 情感指数API
GET /api/v1/sentiment/index
  参数:
    - topic: 主题关键词
    - platform: twitter|weibo|news|all
  响应:
    {
      "topic": "选举",
      "index": {
        "current": -0.35,
        "change_24h": -0.12,
        "trend": "declining"
      },
      "volume": 15000,
      "timestamp": "2026-03-17T06:20:00Z"
    }

# 事件预警API
GET /api/v1/alerts
  响应:
    {
      "alerts": [
        {
          "id": "alert_xxx",
          "severity": "high",
          "type": "sentiment_spike",
          "message": "选举话题负面情感激增20%",
          "related_markets": ["选举市场"],
          "created_at": "2026-03-17T06:15:00Z"
        }
      ]
    }
```
